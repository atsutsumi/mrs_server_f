def addHeader(data)
        packet = ""
        packet << @type_of_mode
        packet << @sender
        packet << @file_format
        packet << data

        return packet
end

          def enqueue?(data)
result = false

                  begin
  # uses unix domain socket
  sock_path = @sock_path
                          enq_sock = UNIXSocket.new(sock_path)
  
  packet = addHeader(data)

  Lgdisit.logger.info(Lgdisit.get_object_id_log(self.object_id) + "Send: " + packet)
                          enq_sock.send data,0
  result = true
                  rescue => exception
  result = false
                          Lgdisit.logger.error(exception)     
                  ensure
                          if !enq_sock.nil?
                                  enq_sock.close 
    Lgdisit.logger.info(Lgdisit.get_object_id_log(self.object_id) + "Socket closed.")
                          end
                  end

return result
          end
