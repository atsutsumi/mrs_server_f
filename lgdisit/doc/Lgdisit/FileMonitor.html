def start
        Lgdisit.logger.info(Lgdisit.get_object_id_log(self.object_id) + "Start monitoring.")

        begin
                fam = Fam::Connection.new
                dir_request = fam.monitor_dir(@monitor_dir)

                # Runs event loop.
                # If files are created and fam events pending exists, process the files.
                while true
                        # to avoid heavy load on CPU
                        sleep(0.1)

                        if fam.pending?
                                ev = fam.next_event

                                case ev.code
                                        when Fam::Event::CREATED 
                                                Lgdisit.logger.info(Lgdisit.get_object_id_log(self.object_id) + "File created: " + ev.file)

                                                # check if the file name matches the filter in the configuration yaml
                                                if !match_file_name_filter?(ev.file)
                                                        next
                                                end

                                                file_path = File.join(@monitor_dir, ev.file)

# check if the file exists
if !File.exists?(file_path)
  Lgdisit.logger.info(Lgdisit.get_object_id_log(self.object_id) + "File might have been removed.")
  next
end

                                                File.open(file_path,"r") do |io|
                                                        io.binmode

                                                        data = io.read

                                                        client = Lgdisit::QueueClient.new(@type_of_mode, @sender, get_file_format(file_path), @sock_path)
                                                        success =  client.enqueue(data)

  # archive the file
  if success && File.exists?(file_path)
    FileUtils.mv(file_path,@archive_path)

    archive_log_message = ""
    archive_log_message << Lgdisit.get_object_id_log(self.object_id)
    archive_log_message << "Archived:"
    archive_log_message << file_path
    archive_log_message << " => "
    archive_log_message << Lgdisit.get_ftp_config["archive_path"]
    Lgdisit.logger.info(archive_log_message)
  end
                                                end

                                        when Fam::Event::DELETED 
                                                # this case is written only for debugging purposes
                                                Lgdisit.logger.debug(Lgdisit.get_object_id_log(self.object_id) + "File deleted: " + ev.file)
                                        when Fam::Event::CHANGED 
                                                # this case is written only for debugging purposes
                                                Lgdisit.logger.debug(Lgdisit.get_object_id_log(self.object_id) + "File modified: " + ev.file)
                                end
                        end
                end
        rescue => exception
                Lgdisit.logger.fatal(exception)     
                Lgdisit.send_error(exception)
        end
        Lgdisit.logger.info( Lgdisit.get_object_id_log(self.object_id) + "Stop monitoring.")
end
