def self.get_config_path(file_path)
        @config_path ||= File.join(get_parent_path(file_path), "config")
        return @config_path
end

def self.get_parent_path(file_path)
        @parent_path ||= File.dirname(File.dirname(File.expand_path(file_path)))
        return @parent_path
end
